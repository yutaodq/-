定义路径别名，要修改两个文件
1- tsconfig.json 文件

        "paths": {
            "zyapp/*": ["src/main/webapp/app/*"],
            "env/*": ["src/environments/*"]
        },
2- webpack.common.js
module.exports = (options) => ({
  resolve: {
    extensions: [".ts", ".js"],
    modules: ["node_modules"],
    alias: {
      zyapp: utils.root("src/main/webapp/app/"),
      env: utils.root("src/environments/"),
      ...rxPaths()
    }
  },
